# Decimal to binary conversion

def decimal_to_binary(num):

    t = bin(num).replace("0b", "")

    

    if(len(t)%4 != 0):

        div = len(t) / 4

        div = int(div)

        counter =(4 * (div + 1)) - len(t)

        for i in range(0, counter):

            t = '0' + t

    return t

# binary to Decimal conversion

def bin2dec(b):

    res=0

    j=0

    for i in range(len(b)-1,-1,-1):

        res += int(b[i])*(2**(j))

        j+=1

    return res

def encryption(plaintext1,round_key_list):

    plaintext1 = hex2bin(plaintext1)

     

    # Initial Permutation

    plaintext1 = initial_permute(plaintext1)

    # Splitting

    left = plaintext1[0:32]

    right = plaintext1[32:64]

    

    #6ROUND DES

    for i in range(0, 6):

        #  Expansion 

        e = expansion(right)

         

        # XOR RoundKey and expansion output

        xor_op = xor_bitwise(e, round_key_list[i])

        

        #S-BOX output

        s_op = ''

        s_i=0

        for i in range(0,48,6):

            s_op += s_box(xor_op[i:i+6],s_i)

            s_i += 1

        

        #permutation

        op_perm = permutation(s_op)

        

         # XOR left and permutation_output

        result = xor_bitwise(left, op_perm)

        

        #NEXT ROUND inputs

        left, right = right, result

        

    # Combination

    combine = left + right

    

    # Final permutation

    cipher_text = final_permute(combine)

    return cipher_text
